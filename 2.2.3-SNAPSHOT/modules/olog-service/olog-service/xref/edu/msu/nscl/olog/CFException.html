7   package edu.msu.nscl.olog;
8   
9   import javax.ws.rs.core.Response;
10  
11  /**
12   * Olog Exception that creates the matching HTTP Response.
13   *
14   * @author Eric Berryman taken from Ralph Lange <Ralph.Lange@bessy.de>
15   */
16  public class CFException extends Exception {
17  
18      private Response.Status status;
19  
20      /**
21       * Creates a new CFException with the specified HTTP return code for this request,
22       * detail message and cause.
23       *
24       * @param status HTTP return code
25       * @param message
26       * @param cause
27       */
28      public CFException(Response.Status status, String message, Throwable cause) {
29          super(message, cause);
30          this.status = status;
31      }
32  
33      /**
34       * Creates a new CFException with the specified HTTP return code for this request.
35       * and detail message.
36       *
37       * @param status HTTP return code
38       * @param message
39       */
40      public CFException(Response.Status status, String message) {
41          super(message);
42          this.status = status;
43      }
44  
45      private String responseMessage() {
46          String msg = "<!DOCTYPE html PUBLIC \"-//W3C//DTD XHTML 1.0 Strict//EN\"" +
47                  " \"http://www.w3.org/TR/xhtml1/DTD/xhtml1-strict.dtd\">" +
48                  "<html><head><title>Olog - Error report</title></head>" +
49                  "<body><h1>HTTP Status " + this.status.getStatusCode() + " - " + this.status.getReasonPhrase() + "</h1><hr/>" +
50                  "<p><b>type</b> Status report</p>" +
51                  "<p><b>message</b></p>" +
52                  "<p><b>description</b>" + getMessage() + "</p>";
53          if (this.getCause() != null) {
54              msg = msg + "<p><b>caused by:</b></p><p>" + this.getCause().getMessage() + "</p>";
55          }
56          return msg + "<hr/><h3>Olog</h3></body></html>";
57      }
58  
59      /**
60       * Returns a HTTP Response object for this exception.
61       * @return HTTP response
62       */
63      public Response toResponse() {
64          return Response.status(status)
65                      .entity(responseMessage())
66                      .build();
67      }
68  
69      /**
70       * Returns the HTTP Response status code for this exception.
71       * @return HTTP response
72       */
73      public Response.Status getResponseStatusString() {
74          return status;
75      }
76  
77      /**
78       * Returns the HTTP Response status code for this exception.
79       * @return HTTP response
80       */
81      public int getResponseStatusCode() {
82          return status.getStatusCode();
83      }
84  }

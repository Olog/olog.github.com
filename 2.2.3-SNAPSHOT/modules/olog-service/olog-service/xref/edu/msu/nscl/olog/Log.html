471     /**
472      * Creates a compact string representation for the log.
473      *
474      * @param data Log to create the string representation for
475      * @return string representation
476      */
477     public static String toLogger(Log data) {
478         Set<Logbook> xl = data.getLogbooks();
479         Set<Tag> xt = data.getTags();
480 
481         return data.getId() + "-v." + data.getVersion() + " : " + "(" + data.getOwner() + "):["
482                 + Logbooks.toLogger(xl)
483                 + Tags.toLogger(xt)
484                 + "]\n";
485     }

94      /**
95       * Creates a compact string representation for the log.
96       *
97       * @param data Log to create the string representation for
98       * @return string representation
99       */
100     public static String toLogger(Logs data) {
101         if (data.getLogs().size() == 0) {
102             return "[None]";
103         } else {
104             StringBuilder s = new StringBuilder();
105             s.append("[");
106             for (Log c : data.getLogs()) {
107                 s.append(Log.toLogger(c) + ",");
108             }
109             s.delete(s.length() - 1, s.length());
110             s.append("]");
111             return s.toString();
112         }
113     }

115     public static String toLogger(List<Log> data) {

7   package edu.msu.nscl.olog;
8   
9   import java.security.Principal;
10  import java.util.Collection;
11  import java.util.Set;
12  import java.util.logging.Level;
13  import java.util.logging.Logger;
14  import javax.naming.InitialContext;
15  
16  /**
17   * Owner (group) membership management.
18   *
19   * @author Ralph Lange <Ralph.Lange@helmholtz-berlin.de>
20   * @author Gabriele Carcassi <carcassi@bnl.gov>
21   */
22  public abstract class UserManager {
23      
24      private static final Logger log = Logger.getLogger(UserManager.class.getName());
25      
26      private ThreadLocal<Principal> user = new ThreadLocal<Principal>();
27      private ThreadLocal<Boolean> hasAdminRole = new ThreadLocal<Boolean>();
28      private ThreadLocal<Collection<String>> groups = new ThreadLocal<Collection<String>>();
29      private ThreadLocal<String> hostName = new ThreadLocal<String>();
30      
31      private static final String defaultUserManager = "edu.msu.nscl.olog.IDUserManager";
32      private static final String userManager;
33      private static UserManager instance;
34      
35      static {
36          String newManager = defaultUserManager;
37          try {
38              newManager = (String) new InitialContext().lookup("olog/userManager");
39              log.log(Level.CONFIG, "Found olog/userManager: {0}", newManager);
40          } catch (Exception ex) {
41              log.log(Level.CONFIG, "Using default olog/userManager: {0}", newManager);
42          }
43          userManager = newManager;
44          
45          try {
46              instance = (UserManager) Class.forName(userManager).newInstance();
47          } catch (ClassNotFoundException ex) {
48              log.log(Level.SEVERE, "Could not find class {0}", userManager);
49          } catch (IllegalAccessException ex) {
50              log.log(Level.SEVERE, "No public constructor for class {0}", userManager);
51          } catch (Exception ex) {
52              log.log(Level.SEVERE, "Public constructor failed for class " + userManager, ex);
53          }
54      }
55  
56      protected UserManager() {
57      }
58  
59      /**
60       * Returns the (singleton) instance of UserManager.
61       *
62       * @return instance of UserManager
63       */
64      public static UserManager getInstance() {
65          if (instance == null)
66              throw new IllegalStateException("UserManager could not be instanced");
67          return instance;
68      }
69      
70      /**
71       * Retrieves the group membership for the given principal.
72       * 
73       * @param user a user
74       * @return the group names
75       */
76      protected abstract Set<String> getGroups(Principal user);
77  
78      /**
79       * Sets the (thread local) user principal to be used in further calls
80       * and retrieves the group information.
81       *
82       * @param user principal
83       * @param isAdmin flag: true = user has Admin role
84       */
85      public void setUser(Principal user, boolean isAdmin) {
86          this.user.set(user);
87          this.hasAdminRole.set(isAdmin);
88          this.groups.set(getGroups(user));
89      }
90  
91      /**
92       * Checks if the user is in the specified <tt>group</tt>.
93       *
94       * @param group name of the group to check membership
95       * @return true if user is a member of <tt>group</tt>
96       */
97      public boolean userIsInGroup(String group) {
98          return group == null ? true : groups.get().contains(group);
99      }
100 
101     /**
102      * Checks if the user has admin role.
103      *
104      * @return true if user is a member of <tt>group</tt>
105      */
106     public boolean userHasAdminRole() {
107         return hasAdminRole.get();
108     }
109 
110     /**
111      * Returns the current user's name.
112      *
113      * @return name of current user
114      */
115     public String getUserName() {
116         return user.get().getName();
117     }
118     
119     /**
120      * Sets the (thread local) user hostName to be used in further calls.
121      *
122      * @param haostName String of host name
123      */
124     public void setHostAddress(String hostName) {
125         this.hostName.set(hostName);
126     }
127     
128     /**
129      * Returns the current user's hostName.
130      *
131      * @return hostName of current user
132      */
133     public String getHostAddress() {
134         return hostName.get();
135     }
136 }
